CREATE TABLE gps(
	gpsid integer DEFAULT nextval('gps_seq'::regclass) NOT NULL,
	latitude float NOT NULL,
	longitude float NOT NULL,
	altitude float NOT NULL
);

CREATE TABLE battery(
	batteryid integer DEFAULT nextval('battery_seq'::regclass) NOT NULL,
	technology varchar(20),
	ispresent integer,
	plugged integer,
	scale integer,
	health integer,
	voltage integer,
	level integer,
	temperature integer,
	status integer
	
);

CREATE TABLE link(
	linkid integer DEFAULT nextval('link_seq'::regclass) NOT NULL,
	count int,
	message_size bigint,
	duration integer,
	speed float NOT NULL,
	port integer NOT NULL,
	ip_address varchar(20) NOT NULL
);

CREATE TABLE throughput(
	throughputid integer DEFAULT nextval('throughput_seq'::regclass) NOT NULL,
	uplinkid integer DEFAULT nextval('link_seq'::regclass) NOT NULL,
	downlinkid integer DEFAULT nextval('link_seq'::regclass) NOT NULL
);

CREATE TABLE device (
    deviceid varchar(20) NOT NULL,
    phonetype varchar(20),
    phonenumber varchar(15) NOT NULL,
    softwareversion varchar(10),
    phonemodel varchar(20),
    androidversion varchar(10),
    phonebrand varchar(20),
    devicedesign varchar(20),
    manufacturer varchar(20),
    productname varchar(20),
    radioversion varchar(20),
    boardname varchar(20)
);

CREATE TABLE network (
    networkid integer DEFAULT nextval('network_seq'::regclass) NOT NULL,
    networkcountry varchar(2),
    networkname varchar(15),
    networktype varchar(10),
    connectiontype varchar(10),
    mobilenetworkinfo text,
    wifistate varchar(20),
    cellid varchar(20),
    celllac varchar(20),
    datastate varchar(30),
    dataactivity varchar(30),
    signalstrength varchar(3)
);

CREATE TABLE sim (
    serialnumber varchar(20) NOT NULL,
    state varchar(20),
    operatorCode varchar(8),
    operatorName varchar(20),
    networkcountry varchar(5)
);

CREATE TABLE measurement (
    measurementid integer DEFAULT nextval('measurement_seq'::regclass) NOT NULL,
    "time" timestamp without time zone NOT NULL,
    deviceid varchar(20) NOT NULL,
    networkid integer DEFAULT nextval('network_seq'::regclass) NOT NULL,
    serialnumber varchar(20) NOT NULL,
    throughputid integer DEFAULT nextval('throughput_seq'::regclass) NOT NULL,
    gpsid integer DEFAULT nextval('gps_seq'::regclass) NOT NULL,
    batteryid integer DEFAULT nextval('battery_seq'::regclass) NOT NULL,
    usageid integer DEFAULT nextval('usage_seq'::regclass) NOT NULL
);

CREATE TABLE usage (
	usageid integer DEFAULT nextval('usage_seq'::regclass) NOT NULL,
	total_sent bigint NOT NULL,
	total_recv bigint NOT NULL,
	mobile_sent bigint NOT NULL,
	mobile_recv bigint NOT NULL
);

CREATE TABLE application (
	name varchar(20) NOT NULL,
	package varchar(50) NOT NULL
);

CREATE TABLE application_use (
	application_useid integer DEFAULT nextval('application_use_seq'::regclass) NOT NULL,
	package varchar(50) NOT NULL,
	total_sent bigint NOT NULL,
	total_recv bigint NOT NULL,
	usageid integer DEFAULT nextval('usage_seq'::regclass) NOT NULL
);


CREATE TABLE ping (
    avg double precision NOT NULL,
    stdev double precision NOT NULL,
    min double precision NOT NULL,
    max double precision NOT NULL,
    scrip varchar(20) NOT NULL,
    dstip varchar(20) NOT NULL,
    "time" timestamp without time zone,
    measurementid integer DEFAULT nextval('measurement_seq'::regclass) NOT NULL,
    pingid integer DEFAULT nextval('ping_seq'::regclass) NOT NULL
);
